Terreno de Ruido

por emm3

Malla 3D (WEBGL) tipo terreno generada con ruido Perlin, reestilizada al tema del sitio: cuerpo verde matrix con las cimas en naranja neón sobre fondo negro. Pieza autónoma con dos LFOs que modulan el relieve y giro lento en Z. Suma un DRON audio-reactivo en Tone.js: un super-saw polifónico toca la secuencia Am–F–Am–G en octavas bajas a través de un LPF estilo Moog con LFOs en el cutoff y la resonancia, mezclado con una voz de ruido blanco filtrada por un BPF (con LFO) cuyo volumen sigue el grado de ruido de la animación. Perilla de volumen máximo y botón de audio on/off con fundido continuo. Basada en Generative Design (cap. M).

🔊 Requiere audio: hacé clic en el canvas para habilitar el sonido.

Controles

  • Audio ON/OFFActiva/desactiva el dron con fundido continuo.
  • Vol. máxPerilla del volumen máximo de las dos voces.
  • arrastrarInclinar/orbitar la cámara.
  • flechas / + −Falloff, octavas (suben el grado de ruido) y zoom.
  • espacio / sNueva semilla / guardar PNG.

20/**
21 * creates a terrain like mesh based on noise values.
22 *
23 * MOUSE
24 * position x/y + left drag   : specify noise input range
25 * position x/y + right drag  : camera controls
26 *
27 * KEYS
28 * arrow up                   : noise falloff +
29 * arrow down                 : noise falloff -
30 * arrow left                 : noise octaves -
31 * arrow right                : noise octaves +
32 * space                      : new noise seed
33 * +                          : zoom in
34 * -                          : zoom out
35 * s                          : save png
36 */

38// ------ mesh ------
39var tileCount;
40var zScale;
41
42// ------ noise ------
43var noiseXRange;
44var noiseYRange;
45var octaves;
46var falloff;
47
48// ------ mesh coloring ------
49var midColor;
50var topColor;
51var bottomColor;
52var threshold;
53
54// ------ mouse interaction ------
55var offsetX;
56var offsetY;
57var clickX;
58var clickY;
59var zoom;
60var rotationX;
61var rotationZ;
62var targetRotationX;
63var targetRotationZ;
64var clickRotationX;
65var clickRotationZ;
66
67function setup() {
68  createCanvas(600, 600, WEBGL);
69  colorMode(HSB, 360, 100, 100);
70  cursor(CROSS);
71
72  // ------ mesh ------
73  tileCount = 50;
74  zScale = 150;
75
76  // ------ noise ------
77  noiseXRange = 10;
78  noiseYRange = 10;
79  octaves = 4;
80  falloff = 0.5;
81
82  // ------ mesh coloring ------
83  topColor = color(0, 0, 100);
84  midColor = color(191, 99, 63);
85  bottomColor = color(0, 0, 0);
86  threshold = 0.30;
87
88  // ------ mouse interaction ------
89  offsetX = 0;
90  offsetY = 0;
91  clickX = 0;
92  clickY = 0;
93  zoom = -300;
94  rotationX = 0;
95  rotationZ = 0;
96  targetRotationX = PI / 3;
97  targetRotationZ = 0;
98}
99
100function draw() {
101  background(0, 0, 100);
102
103  // ------ set view ------
104  push();
105  translate(width * 0.05, height * 0.05, zoom);
106
107  if (mouseIsPressed && mouseButton == RIGHT) {
108    offsetX = mouseX - clickX;
109    offsetY = mouseY - clickY;
110    targetRotationX = min(max(clickRotationX + offsetY / float(width) * TWO_PI, -HALF_PI), HALF_PI);
111    targetRotationZ = clickRotationZ + offsetX / float(height) * TWO_PI;
112  }
113  rotationX += (targetRotationX - rotationX) * 0.25;
114  rotationZ += (targetRotationZ - rotationZ) * 0.25;
115  rotateX(rotationX);
116  rotateZ(-rotationZ);
117
118  // ------ mesh noise ------
119  if (mouseIsPressed && mouseButton == LEFT) {
120    noiseXRange = mouseX / 10;
121    noiseYRange = mouseY / 10;
122  }
123
124  noiseDetail(octaves, falloff);
125  var noiseYMax = 0;
126
127  var tileSizeY = height / tileCount;
128  var noiseStepY = noiseYRange / tileCount;
129
130  for (var meshY = 0; meshY <= tileCount; meshY++) {
131    beginShape(TRIANGLE_STRIP);
132    for (var meshX = 0; meshX <= tileCount; meshX++) {
133
134      var x = map(meshX, 0, tileCount, -width / 2, width / 2);
135      var y = map(meshY, 0, tileCount, -height / 2, height / 2);
136
137      var noiseX = map(meshX, 0, tileCount, 0, noiseXRange);
138      var noiseY = map(meshY, 0, tileCount, 0, noiseYRange);
139      var z1 = noise(noiseX, noiseY);
140      var z2 = noise(noiseX, noiseY + noiseStepY);
141
142      noiseYMax = max(noiseYMax, z1);
143      var interColor;
144      colorMode(RGB);
145      var amount;
146      if (z1 <= threshold) {
147        amount = map(z1, 0, threshold, 0.15, 1);
148        interColor = lerpColor(bottomColor, midColor, amount);
149      } else {
150        amount = map(z1, threshold, noiseYMax, 0, 1);
151        interColor = lerpColor(midColor, topColor, amount);
152      }
153      fill(interColor);
154      vertex(x, y, z1 * zScale);
155      vertex(x, y + tileSizeY, z2 * zScale);
156    }
157    endShape();
158  }
159  pop();
160
161}
162
163function mousePressed() {
164  clickX = mouseX;
165  clickY = mouseY;
166  clickRotationX = rotationX;
167  clickRotationZ = rotationZ;
168}
169
170function keyReleased() {
171  if (keyCode == UP_ARROW) falloff += 0.05;
172  if (keyCode == DOWN_ARROW) falloff -= 0.05;
173  if (falloff > 1.0) falloff = 1.0;
174  if (falloff < 0.0) falloff = 0.0;
175
176  if (keyCode == LEFT_ARROW) octaves--;
177  if (keyCode == RIGHT_ARROW) octaves++;
178  if (octaves < 0) octaves = 0;
179
180  if (keyCode == 187) zoom += 20; // '+'
181  if (keyCode == 189) zoom -= 20; // '-'
182
183  if (key == 's' || key == 'S') saveCanvas(gd.timestamp(), 'png');
184  if (key == ' ') noiseSeed(floor(random(100000)));
185}
