Caverna a Impulsos

por emm3

Un río subterráneo raymarcheado por el que avanzás a empujones. El túnel-caverna no es geometría: cada píxel lanza un rayo que serpentea por un tubo que respira, choca con una superficie de agua y la refracta —una segunda marcha— para ver el fondo a través de ella; una luz se mece por dentro (la esfera que la portaba se quitó, pero su resplandor viaja con ella). Lo distinto es cómo se avanza: el tiempo no corre solo. Cada impulso suma velocidad y la fricción la frena, así un empujón te hace planear unos pasos y varios seguidos aceleran de verdad. Empujás con el botón 'Adelante' (o tocando el lienzo) y, con la cámara encendida, acercando la mano de golpe hacia la lente: el tamaño aparente de la mano hace de sensor de profundidad. Un medidor muestra la fuerza acumulada. Sin cámara se juega igual con el botón y las perillas.

  • InerciaPerilla: cuánto conserva la velocidad (fricción); más = planeás más lejos.

  • CalidadPerilla: pasos del raymarch (más = más detalle y más costo).
  • Reiniciar / rBotón: vuelve el avance y la velocidad a cero.
  • HUD / hMostrar u ocultar el preview de cámara y el estado.

textoEstado() : "");488  };489 490  // --- Física del avance por impulso ----------------------------------------491 492  /**493   * El desplazamiento hacia adelante viene solo de `avance`; el reloj ambiental494   * `uTiempo` agrega la deriva mínima para que el mundo respire en reposo.495   */496  function avanzarFisica() {497    avance += velocidad;498    velocidad *= friccion;499    if (velocidad < 1e-4) velocidad = 0;500  }501 502  /** Reloj ambiental: avance + deriva por tiempo real (agua/luz/cueva/textura). */503  function relojAmbiente(): number {504    return avance + CONFIG.derivaAmbiente * (p.millis() / 1000);505  }506 507  /** Un impulso: suma a la velocidad (acumulable) y enciende el destello del HUD. */508  function impulsar() {509    velocidad = Math.min(velocidad + impulso, velMax);510    flashHasta = p.millis() + CONFIG.flash;511  }512 513  function reiniciar() {514    avance = 0;515    velocidad = 0;516    detector.reiniciar();517  }518 519  // --- Gesto: acercar la mano -----------------------------------------------520 521  /**522   * Deriva el tamaño aparente de la mano (proxy de profundidad) y dispara un523   * impulso en el flanco de acercamiento. `RastreadorManos` no da distancia a la524   * cámara: el tamaño aparente es la distancia muñeca→nudillo medio, que crece525   * al acercar la mano. Se suaviza con lerp antes de pasarlo al detector, o el526   * jitter del landmark dispararía falsos positivos.527   */528  function leerGesto() {529    const manos = usarCamara && rastreador.activo ? rastreador.estado.manos : [];530    if (manos.length === 0) {531      detector.reiniciar();532      habiaMano = false;533      return;534    }535    const mano = manos[0];536    const tam = Math.hypot(537      mano.puntos[0].x - mano.puntos[9].x,538      mano.puntos[0].y - mano.puntos[9].y,539    );540    // Al reaparecer la mano, arrancar el suavizado en la lectura actual para no541    // fabricar un flanco falso desde el valor viejo.542    tamanoSuave = habiaMano543      ? tamanoSuave + (tam - tamanoSuave) * CONFIG.suavizado544      : tam;545    habiaMano = true;546 547    if (detector.actualizar(tamanoSuave, p.millis())) impulsar();548  }549 550  // --- Dibujo ---------------------------------------------------------------551 552  /** Cuad de pantalla completa con el shader de la caverna. */553  function dibujarCaverna() {554    p.push();555    p.shader(shaderCaverna);556    shaderCaverna.setUniform("uResolucion", [p.width, p.height]);557    shaderCaverna.setUniform("uAvance", avance);558    shaderCaverna.setUniform("uTiempo", relojAmbiente());559    shaderCaverna.setUniform("uPasos", pasos);560    shaderCaverna.setUniform("uRoca", rgb(PALETA.roca));561    shaderCaverna.setUniform("uAgua", rgb(PALETA.agua));562    shaderCaverna.setUniform("uEspecular", rgb(PALETA.especular));563    // El tamaño del plano da igual: el vertex shader lo manda a clip space desde564    // la UV, no desde la posición.565    p.plane(p.width, p.height);566    p.pop();567    p.resetShader();568  }569 570  /**571   * Medidor de la fuerza acumulada: una barra que muestra `velocidad/velMax`,572   * para que se vea que el empujón registró y que la inercia sube al insistir.573   * Se dibuja con geometría (no `text()`, que en WEBGL exige fuente cargada).574   */575  function dibujarMedidor() {576    const w = 220;577    const h = 10;578    // WEBGL centra el origen: la esquina inferior izquierda se calcula a mano.579    const x = -p.width / 2 + 12;580    const y = p.height / 2 - 58;581    const destello = p.millis() < flashHasta;582    const frac = velMax > 0 ? Math.min(velocidad / velMax, 1) : 0;583 584    p.push();585    p.noStroke();586    p.fill(colLinea);587    p.rect(x, y, w, h);588    p.fill(destello ? colPico : colMedidor);589    p.rect(x, y, w * frac, h);590    p.pop();591  }592 593  /** Preview de la cámara con el esqueleto de las manos, abajo a la derecha. */594  function dibujarPreview() {595    if (!usarCamara) return;596    const w = CONFIG.preview;597    const x = p.width / 2 - w - 16;598    const y = p.height / 2 - w * 0.75 - 78;599    const destello = p.millis() < flashHasta;600    p.push();601    p.resetShader();602    rastreador.dibujarPreview(x, y, w, {603      linea: PALETA.hudAcento,604      punto: PALETA.hud,605      // El marco parpadea naranja al detectar un empujón por gesto.606      marco: destello ? PALETA.pico : PALETA.hudLinea,607    });608    p.pop();609  }610 611  // --- Controles táctiles ----------------------------------------------------612 613  function crearControles() {614    new Knob(p, {615      etiqueta: "Impulso",616      min: 0.05,617      max: 1,618      valor: impulso,619      paso: 0.01,620      onChange: (v) => {621        impulso = v;622      },623    }).position(12, 10);624 625    new Knob(p, {626      etiqueta: "Inercia",627      min: 0.7,628      max: 0.985,629      valor: friccion,630      paso: 0.005,631      onChange: (v) => {632        friccion = v;633      },634    }).position(76, 10);635 636    new Knob(p, {637      etiqueta: "Vel. máx",638      min: 0.2,639      max: 3,640      valor: velMax,641      paso: 0.05,642      onChange: (v) => {643        velMax = v;644      },645    }).position(140, 10);646 647    new Knob(p, {648      etiqueta: "Calidad",649      min: 24,650      max: 96,651      valor: pasos,652      paso: 1,653      onChange: (v) => {654        pasos = v;655      },656    }).position(204, 10);657 658    // Menú de cámaras: los `label` de `enumerateDevices()` llegan vacíos hasta659    // que hay permiso, por eso el selector trae su propio botón "Buscar".660    selectorCamara = new SelectorCamara(p, {661      ancho: 200,662      onCambio: (deviceId) => {663        // Cambio en caliente: se reabre el stream sin recargar el modelo.664        if (usarCamara) void rastreador.iniciar(deviceId);665      },666    });667    selectorCamara.position(660, 10);668 669    // "Adelante" es la acción central: acento naranja (CTA). void rastreador.iniciar(selectorCamara.valor());683        else rastreador.detener();684      },685    }).position(120, 540);686 687    new Boton(p, {688      etiqueta: "Reiniciar",689      onPress: activo;698      },699    }).position(338, 540);700 701    new Boton(p, {702      etiqueta: "Guardar",703      onPress: () => p.saveCanvas(SLUG, "png"),704    }).position(426, 540);705 706    // El estado va en un panel DOM: en WEBGL, `text()` exige fuente con `loadFont`.707    panelEstado = new PanelEstado(p);708    panelEstado.position(12, 508);709  }710 711  // --- HUD -------------------------------------------------------------------712 713  function textoEstado(): string {714    const comun = `avance: ${avance.toFixed(1)}  vel: ${velocidad.toFixed(2)}`;715    if
