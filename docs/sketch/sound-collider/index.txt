function logConv(posicion, min ,max, minLog, maxLog) {
    var minp = min;
    var maxp = max;
    var minv = Math.log(minLog);
    var maxv = Math.log(maxLog);
    var scale = (maxv-minv) / (maxp-minp);
  
    return Math.exp(minv + scale*(posicion-minp));
