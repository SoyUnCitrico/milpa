Aros Pulsantes

por mndrn.mx

Rejilla de figuras dibujadas con bucles anidados: en cada celda un punto blanco, un aro azul y un cuadrado rojo que laten al unísono. Los tamaños hacen ping-pong, produciendo el pulso. Loop continuo.

Controles

  • sGuardar el canvas como PNG (saveCanvas).

1  let diametro, x,y,aumento, vel, xInicial, yInicial;
2  let size1, size2;
3
4  function setup() {
5    createCanvas(600, 600).parent("sketch-container");
6    // cnv.parent('main__sketch');
7    background(220);
8    rectMode(CENTER);
9    // diametro inicial de las figuras
10    diametro = 30;
11    // posicion de la esquina superior izuiqerda de la figura
12    xInicial = diametro*2.75;
13    yInicial = diametro*2.75;
14    // Variables que sirven para animar las figuras
15    size1 = 0;
16    size2 = 0;
17    aumento = 0.075;
18  }
19
20  function draw() {
21    background(10);
22    
23    // Este for se utiliza para dibujar las figuras a lo largo del eje vertical
24    for(let j = 0; j < diametro*16; j+= diametro*2.5){
25      
26      // Este for se utiliza para dibujar las figuras a lo largo del eje horizontal
27      for(let i = 0; i < diametro*16; i += diametro*2.5){
28        
29        // dibuja un circulo blanco pequeño
30        fill(255);
31        noStroke();
32        ellipse(xInicial + i , yInicial + j, 10) ;   
33        // dibuja un circulo azul
34        noFill();
35        strokeWeight(3);
36        stroke(0,120,255);
37        ellipse(xInicial + i , yInicial + j, diametro*2 + size1) ;      
38        // dibuja un cuadrado rojo
39        stroke(255,0,0);
40        strokeWeight(5);
41        rect(xInicial + i, yInicial + j, diametro + size2)
42        // aumenta la variable que aumenta el tamaño del circulo rojo
43        size1+=aumento;
44        
45      } 
46      // aumenta la variable que aumentael tamaño del cuadrado azul
47      size2+=aumento;
48    }
49    // condicional que cambia la direccion del aumento
50    // si la variable de aumento de la figura 2 es mayor a 50 o menor a 0 la variable de aumento cambia de direccion(aumento o disminucion)
51    if (size2 > 30 || size2 < 0) {
52      aumento *= -1;
53    }
54  }
