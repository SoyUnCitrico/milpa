Flor de Shader

por emm3

Flor de tres cuerpos en WEBGL sobre un fondo de oleaje, cada cuerpo con su textura procedural en GLSL —sin una sola imagen— y el tono ciclando en loop infinito. La corona de pétalos es una malla en forma de dona: un perfil polar de lóbulos (de 3 a 10 pétalos), ahuecado como cuenco, con nervaduras de ruido fractal, bandas iridiscentes que recorren rojos, rosas y violetas pero nunca verde, y un reflejo que persigue a la mano. En el hueco central flota un núcleo esférico de textura celular pulsante; del núcleo baja un tallo de cilindros verdes hacia el suelo, y detrás de todo ondula un mar de olas suaves. Las manos, leídas por la cámara, gobiernan los tres cuerpos: los dedos y la pinza rehacen los pétalos, separar las manos agranda el núcleo, subir la mano alarga el tallo. Y si cerrás los ojos, guarda la imagen. Sin cámara se maneja con las perillas.

Controles

  • CámaraBotón: enciende el rastreo y deja que las manos gobiernen los tres cuerpos.
  • menú de cámaraElegir qué cámara usar; 'Buscar cámaras' pide permiso y muestra los nombres reales.
  • dedosDedos levantados (ambas manos): número de pétalos de la dona, de 3 a 10.
  • pinzaSeparar pulgar e índice: forma del pétalo, de redondo a afilado.
  • puñoAbrir o cerrar el puño: cuánto se ahueca la corona (cuenco).
  • manosSeparar o juntar las dos manos: tamaño del núcleo esférico.
  • alturaSubir o bajar la mano: largo del tallo hacia el suelo.
  • cerrar los ojosCon la cámara encendida, guarda la imagen (detección de parpadeo).
  • perillasPétalos, forma, cuenco, núcleo y tallo — los mismos parámetros sin cámara.
  • tonoCicla solo, en loop infinito (no hay control).
  • HUD / hMostrar u ocultar el preview de cámara y la lectura de gestos.
  • Guardar / sGuardar el canvas como PNG.

a++) {492      const anillo = nodos[a];493      const u = a / CONFIG.anillos; // 0 = borde del hueco, 1 = punta494      // Radio normalizado con hueco central: [0,1] → [agujero, 1].495      const radioNorm = CONFIG.agujero + u * (1 - CONFIG.agujero);496      for (let s = 0; s <= CONFIG.segmentos; s++) {497        const v = s / CONFIG.segmentos; // ángulo normalizado → UV.y498        const theta = v * p.TWO_PI;499        // Ruido sobre el círculo unitario: irregularidad que no rompe la500        // continuidad al cerrar la vuelta (θ=0 y θ=2π coinciden).501        const irregular =502          0.88 +503          0.24 * p.noise(Math.cos(theta) + 1, Math.sin(theta) + 1, semilla);504        const radio = CONFIG.tamano * radioNorm * perfil(theta) * irregular;505        const i = s * 5;506        anillo[i] = radio * Math.cos(theta);507        anillo[i + 1] = radio * Math.sin(theta);508        // Cuenco: el borde cae hacia atrás con el cuadrado del radio.509        anillo[i + 2] = -actual.curvatura * CONFIG.tamano * u * u;510        anillo[i + 3] = u;511        anillo[i + 4] = v;512      }513    }514  }515 516  /** Cose los anillos de dos en dos con tiras de triángulos, bajo el shader FLOR. */517  function dibujarCorona(tiempo: number) {518    p.push();519    p.shader(shaderFlor);520    shaderFlor.setUniform("uTiempo", tiempo);521    shaderFlor.setUniform("uTono", tono);522    shaderFlor.setUniform("uBrillo", CONFIG.brillo);523    shaderFlor.setUniform("uDetalle", 3 + actual.agudeza * 2.5);524    shaderFlor.setUniform("uOndulacion", CONFIG.ondulacion);525    shaderFlor.setUniform("uLuz", [luzX, luzY]);526    // La corona ya no usa uPaleta*: tiene su propia paleta sin verde en el shader.527    for (let a = 0; a < CONFIG.anillos; a++) {528      const interior = nodos[a];529      const exterior = nodos[a + 1];530      p.beginShape(p.TRIANGLE_STRIP);531      for (let s = 0; s <= CONFIG.segmentos; s++) {532        const i = s * 5;533        p.vertex(interior[i], interior[i + 1], interior[i + 2], interior[i + 3], interior[i + 4]);534        p.vertex(exterior[i], exterior[i + 1], exterior[i + 2], exterior[i + 3], exterior[i + 4]);535      }536      p.endShape();537    }538    p.pop();539  }540 541  // --- Cuerpo 2: núcleo esférico --------------------------------------------542 543  /** Radio del núcleo, escalado sobre el hueco de la dona. */544  function radioNucleo(): number {545    return CONFIG.tamano * CONFIG.agujero * actual.nucleo;546  }547 548  /**549   * Esfera en el hueco central. Va centrada (no empujada al frente): el apilado550   * por capas de `p.draw` hace que los pétalos la cubran por fuera del hueco, y551   * un ligero retroceso en Z la niega dentro del cuenco.552   */553  function dibujarNucleo(tiempo: number) {554    const r = radioNucleo();555    p.push();556    p.translate(0, 0, -r * 0.2);557    p.shader(shaderNucleo);558    shaderNucleo.setUniform("uTiempo", tiempo);559    shaderNucleo.setUniform("uTono", tono);560    shaderNucleo.setUniform("uPaletaA", [...CONFIG.paletaA]);561    shaderNucleo.setUniform("uPaletaB", [...CONFIG.paletaB]);562    shaderNucleo.setUniform("uPaletaC", [...CONFIG.paletaC]);563    p.sphere(r, 36, 24);564    p.pop();565  }566 567  // --- Cuerpo 3: tallo -------------------------------------------------------568 569  /**570   * Pila de cilindros que baja del núcleo hacia el suelo (+y). Cada segmento se571   * adelgaza hacia la punta y se desvía un poco en X/Z (`sin` acumulado) para572   * que el tallo caiga con una curva orgánica en vez de recto.573   */574  function dibujarTallo() {575    const seg = CONFIG.talloSeg;576    const largoSeg = actual.tallo / seg;577    const rBase = radioNucleo() * 0.5;578    const rPunta = rBase * 0.45;579    const t = p.millis() / 1000;580 581    p.push();582    p.shader(shaderTallo);583    shaderTallo.setUniform("uTiempo", t);584    // Arranca por debajo del núcleo y ligeramente atrás, para que salga desde585    // detrás de la flor (los pétalos y el núcleo lo tapan en su nacimiento).586    p.translate(0, radioNucleo() * 0.6, -radioNucleo() * 0.4);587    for (let i = 0; i < seg; i++) {588      const frac = i / (seg - 1);589      const radio = p.lerp(rBase, rPunta, frac);590      // Curva suave del tallo: la desviación crece hacia la punta.591      const desvio = Math.sin(i * 0.5 + t * 0.6) * largoSeg * 0.16 * frac;592      p.push();593      p.translate(desvio, largoSeg * 0.5, 0);594      // El cilindro de p5 va por el eje Y: sin rotación ya cae hacia el suelo.595      p.cylinder(radio, largoSeg * 1.04, 16, 1);596      p.pop();597      p.translate(desvio, largoSeg, 0);598    }599    p.pop();600  }601 602  // --- Gestos → parámetros ---------------------------------------------------603 604  /**605   * Vocabulario de gestos repartido entre los tres cuerpos. Cada gesto se606   * aplica solo si su mano está en cuadro: con una mano ya se controla la607   * pieza, con dos el control es completo.608   */609  function leerGestos() {610    const { manos, separacion } = rastreador.estado;611    if (manos.length === 0) return;612 613    // Dedos levantados (ambas manos) → número de pétalos.614    const dedos = manos.reduce((suma, m) => suma + m.dedos, 0);615    objetivo.petalos = p.map(616      dedos,617      0,618      manos.length * 5,619      CONFIG.petalosMin,620      CONFIG.petalosMax,621      true,622    );623 624    // Pinza pulgar-índice → agudeza del pétalo (redondo ↔ afilado).625    objetivo.agudeza = p.map(626      manos[0].pinza,627      0.05,628      0.9,629      CONFIG.agudezaMin,630      CONFIG.agudezaMax,631      true,632    );633 634    // Puño cerrado ↔ mano abierta → curvatura del cuenco.635    const apertura =636      manos.reduce((suma, m) => suma + m.apertura, 0) / manos.length;637    objetivo.curvatura = p.map(apertura, 0, 1, 0.7, 0.12, true);638 639    // Separación de las manos → tamaño del núcleo esférico (con una mano, su640    // propia extensión).641    const nucleoGesto = separacion ?? manos[0].apertura * 0.7;642    objetivo.nucleo = p.map(643      nucleoGesto,644      0.1,645      0.75,646      CONFIG.nucleoMin,647      CONFIG.nucleoMax,648      true,649    );650 651    // Altura de la mano → largo del tallo (mano arriba, tallo largo).652    objetivo.tallo = p.map(653      1 - manos[0].centro.y,654      0,655      1,656      CONFIG.talloMin,657      CONFIG.talloMax,658      true,659    );660 661    // El especular de los pétalos sigue a la mano.662    luzX = p.constrain(manos[0].centro.x, 0, 1);663    luzY = p.constrain(manos[0].centro.y, 0, 1);664  }665 666  // ---
