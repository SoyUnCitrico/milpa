Ondas Moduladas

por Generative Design

Un oscilador modulado en amplitud: la señal de información (seno, verde) y la portadora (coseno, verde tenue) se multiplican para dar la onda combinada, en naranja neón sobre fondo negro. La fase se anima sola de forma continua, así la onda se desplaza suavemente. Estudio de modulación del libro Generative Design (cap. M).

Controles

  • i / cMostrar/ocultar señal de información / portadora.
  • 1 / 2 · 7 / 8Frecuencia · frecuencia de modulación.
  • ← / → · sEmpujar la fase (se anima sola) · guardar PNG.

20/**
21 * draws an amplitude modulated oscillator
22 *
23 * KEYS
24 * i                 : toggle draw info signal
25 * c                 : toggle draw carrier signal
26 * 1/2               : info signal frequency -/+
27 * arrow left/right  : info signal phi -/+
28 * 7/8               : carrier signal frequency -/+ (modulation frequency)
29 * s                 : save png
30 */
31'use strict';
32
33var sketch = function(p) {
34
35  var pointCount = 600;
36  var freq = 2;
37  var phi = 0;
38  var modFreq = 12;
39
40  var drawFrequency = true;
41  var drawModulation = true;
42  var drawCombination = true;
43
44  var angle;
45  var y;
46
47  p.setup = function() {
48    p.createCanvas(p.windowWidth,800);
49    p.noFill();
50    pointCount = p.width;
51  };
52
53  p.draw = function() {
54    p.background(255);
55    p.strokeWeight(1);
56
57    p.translate(0, p.height / 2);
58
59    // draw oscillator with freq and phi
60    if (drawFrequency) {
61      p.beginShape();
62      for (var i = 0; i <= pointCount; i++) {
63        angle = p.map(i, 0, pointCount, 0, p.TAU);
64        y = p.sin(angle * freq + p.radians(phi));
65        y *= p.height / 4;
66        p.vertex(i,y);
67      }
68      p.endShape();
69    }
70
71    // draw oscillator with modFreq
72    if (drawModulation) {
73      p.stroke(0,130,164,128);
74      p.beginShape();
75      for (var i = 0; i <= pointCount; i++) {
76        angle = p.map(i, 0, pointCount, 0, p.TAU);
77        y = p.cos(angle * modFreq);
78        y *= p.height / 4;
79        p.vertex(i,y);
80      }
81      p.endShape();
82    }
83
84    // draw both combined
85    p.stroke(0);
86    p.strokeWeight(2);
87    p.beginShape();
88    for (var i = 0; i <= pointCount; i++) {
89      angle = p.map(i, 0, pointCount, 0, p.TAU);
90      var info = p.sin(angle * freq + p.radians(phi));
91      var carrier = p.cos(angle * modFreq);
92      y = info * carrier;
93      y *= p.height / 4;
94      p.vertex(i,y);
95    }
96    p.endShape();
97  };
98
99  p.keyPressed = function() {
100    if (p.key == 's' || p.key == 'S') p.saveCanvas(gd.timestamp(), 'png');
101
102    if (p.key == 'i' || p.key == 'I') drawFrequency = !drawFrequency;
103    if (p.key == 'c' || p.key == 'C') drawModulation = !drawModulation;
104
105    if (p.key == '1') freq--;
106    if (p.key == '2') freq++;
107    freq = p.max(freq,1);
108
109    if (p.keyCode == p.LEFT_ARROW) phi -= 15;
110    if (p.keyCode == p.RIGHT_ARROW) phi += 15;
111
112    if (p.key == '7') modFreq--;
113    if (p.key == '8') modFreq++;
114    modFreq = p.max(modFreq,1);
115
116    console.log('freq: ' + freq + ', phi: ' + phi + ', modFreq: ' + modFreq);
117  };
118
119};
120
121var myp5 = new p5(sketch);
