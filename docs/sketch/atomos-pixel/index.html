Átomos de Píxel

por emm3

Port a p5 del sketch de Processing pixelAtomProject: captura una imagen (webcam o archivo), la descompone en partículas de color que conservan el píxel original y las anima con 15 modos —steering behaviors (seguir, deambular, manada, vórtice, explosión), composiciones generativas (líneas, círculos, rectángulos, hue-shift) y colonización de espacios— sobre 7 tipos de flow field con transiciones suaves, 9 formas de partícula (incluido ASCII/Matrix) y estelas/mezcla aditiva. Sin el control OSC del original.

119void play() {
120  if (campo != null) campo.update();   // avanza la transición de flow field, si la hay
121  applyBackground();                   // limpia / estela según RenderSettings
122  if (state.showField && campo != null) campo.mostrarCampo();
123
124  Animation animation = animations.get(state.mode);
125  if (animation == null) return;
126
127  // enter() se ejecuta una sola vez, el primer frame en que el modo se activa.
128  if (state.mode != lastMode) {
129    ensureClusterForMode(state.mode);    // FLOCK usa su propio cellSize
130    ensureComposerForMode(state.mode);   // HUESHIFT usa cellSize; resto composeCellSize
131    animation.enter(cluster, state);
132    lastMode = state.mode;
133  }
134
135  // Fuerzas de explosión: se acumulan ANTES de integrar, así se suman a la
136  // fuerza del comportamiento activo en el mismo frame.
137  if (cluster != null) cluster.applyRepulsors(repulsors);
138
139  // La mezcla aditiva hace que las partículas brillen al superponerse.
140  if (render.additive) blendMode(ADD);
141  animation.update(cluster, campo, state);
142  if (render.additive) blendMode(BLEND);   // restaurar para el resto del dibujo
143
144  updateRepulsors();   // dibuja y envejece las explosiones activas
145}

147// Dibuja, envejece y elimina los repulsores (explosiones) ya consumidos.
148void updateRepulsors() {
149  for (int i = repulsors.size() - 1; i >= 0; i--) {
150    Repulsor r = repulsors.get(i);
151    r.draw();
152    r.age();
153    if (r.dead()) repulsors.remove(i);
154  }
155}

173// Reconstruye el cluster si el modo necesita un cellSize distinto al actual.
174// FLOCK usa flockCellSize (menos agentes, más nítido); el resto usa cellSize.
175// Solo reconstruye cuando el tamaño objetivo cambia, así alternar entre modos
176// del mismo tamaño conserva las posiciones de las partículas.
177void ensureClusterForMode(AnimationMode mode) {
178  if (img == null) return;
179  int target = (mode == AnimationMode.FLOCK) ? flockCellSize : cellSize;
180  if (cluster == null || cluster.cellSize != target) {
181    cluster = makeCluster(target);
182  }
183}

185// Reconstruye el composer con el cellSize adecuado al modo de composición:
186// HUESHIFT usa el cellSize normal (mosaico más fino); LINES/CIRCLES/RECTS usan
187// composeCellSize. Los demás modos no usan el composer (no reconstruye).
188void ensureComposerForMode(AnimationMode mode) {
189  if (img == null) return;
190  int target;
191  if (mode == AnimationMode.HUESHIFT) {
192    target = cellSize;
193  } else if (mode == AnimationMode.LINES || mode == AnimationMode.CIRCLES
194             || mode == AnimationMode.RECTS) {
195    target = composeCellSize;
196  } else {
197    return;
198  }
199  if (composer == null || composer.cellSize != target) {
200    composer = new PixelComposer(img, target);
201  }
202}

2173    int count = 0;
2174    // For every boid in the system, check if it's too close
2175    for (PixelAtom other : boids) {

2177      // If the distance is greater than 0 and less than an arbitrary amount (0 when you are yourself)

2180        PVector diff = PVector.sub(position, other.position);

2195      steer.normalize();

2476    for (int i = 0; i < gcols * grows; i++) buckets.add(new ArrayList<PixelAtom>());
