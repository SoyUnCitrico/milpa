Audio Graf

por mdrn.mx

Generador y analizador de audio: un oscilador controlado por sliders de frecuencia (escala logarítmica) y volumen, con su espectro FFT y forma de onda dibujados en tiempo real. Requiere activar el audio.

🔊 Requiere audio: hacé clic en el canvas para habilitar el sonido.

Controles

  • slidersFrecuencia (log) y volumen.
  • 1–4Onda: seno / triángulo / cuadrada / sierra.
  • p / sArrancar / parar el oscilador.
  • lAlternar escala del espectro (lin/log).

128function logslider(position, min ,max, minLog, maxLog) {
129  var minp = min;
130  var maxp = max;
131  var minv = Math.log(minLog);
132  var maxv = Math.log(maxLog);
133  var scale = (maxv-minv) / (maxp-minp);
134
135  return Math.exp(minv + scale*(position-minp));
136}

139function autoCorrelate(buffer) {
140  var newBuffer = [];
141  var nSamples = buffer.length;
142
143  var autocorrelation = [];
144
145  // center clip removes any samples under 0.1
146  if (centerClip) {
147    var cutoff = 0.1;
148    for (var i = 0; i < buffer.length; i++) {
149      var val = buffer[i];
150      buffer[i] = Math.abs(val) > cutoff ? val : 0;
151    }
152  }
153
154  for (var lag = 0; lag < nSamples; lag++){
155    var sum = 0; 
156    for (var index = 0; index < nSamples; index++){
157      var indexLagged = index+lag;
158      if (indexLagged < nSamples){
159        var sound1 = buffer[index];
160        var sound2 = buffer[indexLagged];
161        var product = sound1 * sound2;
162        sum += product;
163      }
164    }
165
166    // average to a value between -1 and 1
167    newBuffer[lag] = sum/nSamples;
168  }
169
170  if (bNormalize){
171    var biggestVal = 0;
172    for (var index = 0; index < nSamples; index++){
173      if (abs(newBuffer[index]) > biggestVal){
174        biggestVal = abs(newBuffer[index]);
175      }
176    }
177    for (var index = 0; index < nSamples; index++){
178      newBuffer[index] /= biggestVal;
179    }
180  }
181
182  return newBuffer;
183}
