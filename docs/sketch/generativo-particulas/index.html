Generativo Partículas

por emm3

Campo de flujo de ruido Perlin (Coding Train): 300 partículas siguen el campo recalculado cada frame y dejan estelas de color que derivan lentamente en el tiempo y rebotan en los bordes. Paleta violeta-magenta-cian neón sobre fondo casi negro, con el color de cada partícula elegido por distribución gaussiana centrada en el violeta medio de la paleta. El cursor atrae (o repele, con clic) a las partículas cercanas. Reusa la biblioteca Particle extendida con el modo 'estela'.

Controles

  • mouseEl cursor atrae a las partículas cercanas; el radio de influencia es local.
  • clicInvierte el modo del cursor entre atraer y repeler.
  • s / SGuardar el canvas como PNG (saveCanvas).

76class Particle {
77    constructor() {
78      this.pos = createVector(random(250, 350), random(250, 350));
79      this.vel = createVector(0, 0);
80      this.acc = createVector(0, 0);
81      this.maxspeed = 4;
82      this.prevPos = this.pos.copy();
83      this.r = 0;
84      this.g = 0;
85      this.b = 255;
86      this.alpha = 255;
87      this.stroke = random(1, 5);
88    }
89  
90    finished() {
91      return this.alpha < 0;
92    }
93  
94    update() {
95      this.vel.add(this.acc);
96      this.vel.limit(this.maxspeed);
97      this.pos.add(this.vel);
98      this.acc.mult(0);
99      this.alpha -= 0.08;
100      this.r += 0.1;
101      this.g += 0.1;
102      this.b -= 0.1;
103    }
104  
105    follow(vectors) {
106      var x = floor(this.pos.x / scl);
107      var y = floor(this.pos.y / scl);
108      var index = x + y * cols;
109      var force = vectors[index];
110      this.applyForce(force);
111    }
112  
113    applyForce(force) {
114      this.acc.add(force);
115    }
116  
117    show() {
118      stroke(this.r, this.g, this.b, this.alpha);
119      strokeWeight(this.stroke);
120      line(this.pos.x, this.pos.y, this.prevPos.x, this.prevPos.y);
121      this.updatePrev();
122    }
123  
124    updateRgb() {
125      if (this.r > 10) {
126        this.r -= random(0.1, 0.2);
127      }
128      if (this.r == 0) {
129        this.r += random(0.1, 0.2);
130      }
131  
132      if (this.g > 10) {
133        this.g -= random(0.1, 0.2);
134      }
135      if (this.g == 0) {
136        this.g += random(0.1, 0.2);
137      }
138  
139      if (this.b == 0) {
140        this.b += random(0.1, 0.3);
141      }
142      if (this.b == 255) {
143        this.b -= random(0.1, 0.3);
144      }
145    }
146  
147  
148  
149    updatePrev() {
150      this.prevPos.x = this.pos.x;
151      this.prevPos.y = this.pos.y;
152    }
153  
154    edges() {
155      if (this.pos.x > width || this.pos.x < 0) {
156        this.vel.x = (-1) * this.vel.x;
157      }
158      if (this.pos.y > height || this.pos.y < 0) {
159        this.vel.y = (-1) * this.vel.y;
160      }
161    }
