Malla Armónica

por Generative Design

Una figura de Lissajous con sus puntos cercanos conectados: la malla de conexiones se dibuja en naranja neón (con desvanecimiento por distancia) y los nodos de la curva en verde matrix, sobre fondo oscuro. La fase rota sola de forma continua y suave, y un texto en naranja muestra frecuencias, modulación y fase. Del libro Generative Design (cap. M).

Controles

  • pPausar/continuar la rotación de la fase.
  • 1-4 · 7-0Frecuencias X/Y y de modulación.
  • ← / →Empujar la fase (se anima sola).
  • sGuardar el canvas como PNG.

20/**
21 * draw lissajous figures with all points connected
22 *
23 * KEYS
24 * 1/2               : frequency x -/+
25 * 3/4               : frequency y -/+
26 * arrow left/right  : phi -/+
27 * 7/8               : modulation frequency x -/+
28 * 9/0               : modulation frequency y -/+
29 * s                 : save png
30 */
31'use strict';
32
33var sketch = function(p) {
34
35  var pointCount = 1000;
36  var lissajousPoints = [];
37  var freqX = 4;
38  var freqY = 7;
39  var phi = 15;
40
41  var modFreqX = 3;
42  var modFreqY = 2;
43
44  var lineWeight = 0.1;
45  var lineColor;
46  var lineAlpha = 50;
47
48  var connectionRadius = 100;
49  var connectionRamp = 6;
50
51  p.setup = function() {
52    p.createCanvas(800,800);
53    p.colorMode(p.RGB, 255, 255, 255, 100);
54    p.noFill();
55
56    lineColor = p.color(0, 50);
57
58    calculateLissajousPoints();
59    drawLissajous();
60  };
61
62  function calculateLissajousPoints() {
63    for (var i = 0; i <= pointCount; i++) {
64      var angle = p.map(i, 0, pointCount, 0, p.TAU);
65
66      var x = p.sin(angle * freqX + p.radians(phi)) * p.cos(angle * modFreqX);
67      var y = p.sin(angle * freqY) * p.cos(angle * modFreqY);
68      x *= p.width / 2 - 30;
69      y *= p.height / 2 - 30;
70
71      lissajousPoints[i] = p.createVector(x,y);
72    }
73  }
74
75  function drawLissajous() {
76    p.background(255);
77    p.strokeWeight(lineWeight);
78    p.push();
79    p.translate(p.width / 2, p.height / 2);
80
81    for (var i1 = 0; i1 < pointCount; i1++) {
82      for (var i2 = 0; i2 < i1; i2++) {
83        var d = lissajousPoints[i1].dist(lissajousPoints[i2]);
84        var a = p.pow(1 / (d / connectionRadius + 1), 6);
85        if (d <= connectionRadius) {
86          p.stroke(lineColor, a * lineAlpha);
87          p.line(
88            lissajousPoints[i1].x,
89            lissajousPoints[i1].y,
90            lissajousPoints[i2].x,
91            lissajousPoints[i2].y
92          );
93        }
94      }
95    }
96    p.pop();
97  }
98
99  p.keyPressed = function() {
100    if (p.key == 's' || p.key == 'S') p.saveCanvas(gd.timestamp(), 'png');
101
102    if (p.key == '1') freqX--;
103    if (p.key == '2') freqX++;
104    freqX = p.max(freqX,1);
105
106    if (p.key == '3') freqY--;
107    if (p.key == '4') freqY++;
108    freqY = p.max(freqY,1);
109
110    if (p.keyCode == p.LEFT_ARROW) phi -= 15;
111    if (p.keyCode == p.RIGHT_ARROW) phi += 15;
112
113    if (p.key == '7') modFreqX--;
114    if (p.key == '8') modFreqX++;
115    modFreqX = p.max(modFreqX,1);
116
117    if (p.key == '9') modFreqY--;
118    if (p.key == '0') modFreqY++;
119    modFreqY = p.max(modFreqY,1);
120
121    calculateLissajousPoints();
122    drawLissajous();
123
124    console.log('freqX: ' + freqX + ', freqY: ' + freqY + ', phi: ' + phi + ', modFreqX: ' + modFreqX + ', modFreqY: ' + modFreqY);
125  };
126
127};
