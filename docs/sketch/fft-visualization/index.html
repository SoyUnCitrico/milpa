Visualización FFT

por mndrn.mx

Visualizador de audio sobre una canción: dibuja la autocorrelación de la forma de onda y un espectro de frecuencias en disposición polar, en HSB. Botones de Play/Stop/Random, sliders de volumen y velocidad, y la tecla t alterna entre la canción y el micrófono.

🔊 Requiere audio: hacé clic en el canvas para habilitar el sonido.

Controles

  • Play / Stop / RandomReproducir, detener o saltar en la canción.
  • slidersVolumen y velocidad de reproducción.
  • tAlternar entre la canción y el micrófono.
  • sGuardar el canvas como PNG (saveCanvas).

88function ilustrar() {
89  background(191, 98, 15, 0.2);
90  // Correlation, dibuja las curva amarillas de la parte superior
91  let timeDomain = fft.waveform(1024, 'float32');
92  let corrBuff = autoCorrelate(timeDomain);
93  stroke(color_4);
94  noFill();
95  beginShape();
96  for (let i = 0; i < corrBuff.length; i++) {
97    let w = map(i, 0, corrBuff.length, 0, width);
98    let h = map(corrBuff[i], -1, 1, 150, 0);
99    curveVertex(w, h);
100  }
101  endShape();
102  
103  beginShape();
104  for (let i = 0; i < corrBuff.length; i++) {
105    let w = map(i, 0, corrBuff.length, width, 0);
106    let h = map(corrBuff[i], -1, 1, 150, 0);
107    curveVertex(w, h);
108  }
109  endShape();  
110  noFill();
111  translate(width/2, height*3/4);
112  
113  /////////////////// ESPECTRO DE FRECUENCIAS
114  let spectrum = fft.analyze();
115  beginShape();
116  for(let i = spectrum.length / 4; i < spectrum.length - 9; i++) {
117    let r = map(spectrum[i], 0, 255, 10, width*3/4);
118    let angle = map(i, spectrum.length / 4, spectrum.length - 9, 150, 390);
119    let x = r * cos(angle);
120    let y = r * sin(angle);
121    nZstroke = noise(nZstroke);
122    strokeWeight(3 * nZstroke);
123    stroke(color_1);
124    fill(color_2);
125    if(i >= spectrum.length/2 && i <= spectrum.length/2 + spectrum.length/16 + spectrum.length/32) {
126      // noStroke();
127      strokeWeight(5 * nZstroke);
128      stroke(45, 91, 95, 0.5);
129      point(-x,-y/3);  
130      // rect(-x,-y/2.5,r/10);  
131      stroke(color_2);
132      fill(color_1);
133    }
134    rect(x,y,r/10);
135    nZstroke += aumentoStroke;
136    }
137  fill(color_back);
138  strokeWeight(2*random());
139  stroke(color_3);
140  ellipse(0,0,map(spectrum[1],0,255,10,width/5));
141  endShape();
142  infoText();
143}

145function autoCorrelate(buffer) {
146  let newBuffer = [];
147  let nSamples = buffer.length;
148
149  let autocorrelation = [];
150
151  if (centerClip) {
152    let cutoff = 0.1;
153    for (let i = 0; i < buffer.length; i++) {
154      let val = buffer[i];
155      buffer[i] = Math.abs(val) > cutoff ? val : 0;
156    }
157  }
158
159  for (let lag = 0; lag < nSamples; lag++){
160    let sum = 0; 
161    for (let index = 0; index < nSamples; index++){
162      let indexLagged = index+lag;
163      if (indexLagged < nSamples){
164        let sound1 = buffer[index];
165        let sound2 = buffer[indexLagged];
166        let product = sound1 * sound2;
167        sum += product;
168      }
169    }
170
171    newBuffer[lag] = sum/nSamples;
172  }
173
174  if (bNormalize){
175    let biggestVal = 0;
176    for (let index = 0; index < nSamples; index++){
177      if (abs(newBuffer[index]) > biggestVal){
178        biggestVal = abs(newBuffer[index]);
179      }
180    }
181    for (let index = 0; index < nSamples; index++){
182      newBuffer[index] /= biggestVal;
183    }
184  }
185
186  return newBuffer;
187}
