class Player {
  constructor(x, y, rad, col) {
    this.x = x;
    this.y = y;
    this.radio = rad;
    this.color = col;
  }

  draw() {
    contextCanvas.beginPath();
    contextCanvas.arc(this.x, this.y, this.radio, 0, Math.PI * 2, true);
    contextCanvas.fillStyle = this.color;
    contextCanvas.fill();
  }

class Proyectil{
    constructor(x,y,rad,col,vel) {

      contextCanvas.beginPath();
      contextCanvas.arc(this.x, this.y, this.radio, 0, Math.PI * 2, true);
      contextCanvas.fillStyle = this.color;
      contextCanvas.fill();
    }
