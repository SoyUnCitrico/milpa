Líneas de Humo

por Generative Design

Miles de agentes siguen un campo de ruido Perlin 3D trazando líneas que se acumulan en contornos que fluyen, como humo o curvas de nivel. Reestilizado al tema del sitio: fondo oscuro con la mayoría de trazos en verde matrix y algunos destellos en naranja neón. Al hacer click, el número de agentes (1000–5000) se ajusta según la posición horizontal del mouse. Reusa la biblioteca AgenteHumo. Del libro Generative Design (cap. M).

Controles

  • clickNº de agentes (1000–5000) según mouseX.
  • 1 / 2Alternar el modo de ruido.
  • cNueva semilla de ruido.
  • borrar / sLimpiar la pantalla / guardar PNG.

21 * noise values (noise 3d) are used to animate a bunch of agents.
22 *
23 * KEYS
24 * 1-2                 : switch noise mode
25 * space               : new noise seed
26 * backspace           : clear screen
27 * s                   : save png
28 */
29'use strict';
30
31var sketch = function(p) {
32
33  var agents = [];
34  var agentCount = 4000;
35  var noiseScale = 100;
36  var noiseStrength = 10;
37  var noiseZRange = 0.4;
38  var noiseZVelocity = 0.01;
39  var overlayAlpha = 10;
40  var agentAlpha = 90;
41  var strokeWidth = 0.3;
42  var drawMode = 1;
43
44  p.setup = function() {
45    p.createCanvas(p.windowWidth, p.windowHeight);
46
47    for (var i = 0; i < agentCount; i++) {
48      agents[i] = new Agent(noiseZRange);
49    }
50  };
51
52  p.draw = function() {
53    p.fill(255, overlayAlpha);
54    p.noStroke();
55    p.rect(0, 0, p.width, p.height);
56
57    // Draw agents
58    p.stroke(0, agentAlpha);
59    for (var i = 0; i < agentCount; i++) {
60      if (drawMode == 1) {
61        agents[i].update1(strokeWidth, noiseScale, noiseStrength, noiseZVelocity);
62      } else {
63        agents[i].update2(strokeWidth, noiseScale, noiseStrength, noiseZVelocity);
64      }
65    }
66  };
67
68  p.keyReleased = function() {
69    if (p.key == 's' || p.key == 'S') p.saveCanvas(gd.timestamp(), 'png');
70    if (p.key == '1') drawMode = 1;
71    if (p.key == '2') drawMode = 2;
72    if (p.key == ' ') {
73      var newNoiseSeed = p.floor(p.random(10000));
74      console.log('newNoiseSeed', newNoiseSeed);
75      p.noiseSeed(newNoiseSeed);
76    }
77    if (p.keyCode == p.DELETE || p.keyCode == p.BACKSPACE) p.background(255);
78  };
79
80};
